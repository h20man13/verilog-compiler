847 Statement* const RecursiveDescent::parse_statement(SymbolTable* const table){
848  Position start = get_start();
849  if (will_match(tok::IF)) return parse_if_statement(table);
850  else if (will_match(tok::CASE)) return parse_case_statement(table);
851  else if (will_match(tok::CASEZ)) return parse_casez_statement(table);
852  else if (will_match(tok::CASEX)) return parse_casex_statement(table);
853  else if (will_match(tok::FOREVER)) return parse_forever_statement(table);
854  else if (will_match(tok::REPEAT)) return parse_repeat_statement(table);
855  else if (will_match(tok::WHILE)) return parse_while_statement(table);
856  else if (will_match(tok::FOR)) return parse_for_statement(table);
857  else if (will_match(tok::WAIT)) return parse_wait_statement(table);
858  else if (will_match(tok::BEGIN)) return parse_sequential_block(table);
859  else if (will_match(tok::DOLLAR)) { //system tasks
860  skip();
861  Token ident = match(tok::IDENTIFIER);
862  std::list<Expression* const> exp_list;
863  if (will_match(tok::LPAR)){
864  eat();
865  if (!will_match(tok::RPAR)) {
866  do{
867  Expression* const exp = parse_expression(table);
868  exp_list.push_back(exp);
869  } while(eat_if_yummy(tok::COMMA));
870  }
871  match(tok::RPAR, REPAIR);
872  }
873  match(tok::SEMI, REPAIR);
874  return new SystemTaskCall(start, ident.get_lexeme(), exp_list);
875  } else if (will_match(tok::LCURL)){
876  Concatenation* const concat = parse_concatenation(table);
877  return parse_procedural_assignment(concat, table);
878  } else if(will_match(tok::IDENTIFIER)){
879  Token ident = skip();
880  std::list<Expression* const> exp_list;
881  if(will_match(tok::LPAR)){
882  std::list<Expression* const> exp_list;
883  eat();
884  if (!will_match(tok::RPAR)) {
885  do{
886  Expression *const exp = parse_expression(table);
887  exp_list.push_back(exp);
888  }
889  while (eat_if_yummy(tok::COMMA));
890  }
891  match(tok::RPAR, REPAIR);
892  match(tok::SEMI, REPAIR);
893  return new TaskCall(start, ident.get_lexeme(), exp_list);
894  } else if(will_match(tok::SEMI)){
895  eat();
896  return new TaskCall(start, ident.get_lexeme(), exp_list);
897  } else if (will_match(tok::LBRACK)){
898  eat();
899  Expression* const exp1 = parse_expression(table);
900  if(will_match(tok::COLON)){
901  ConstantExpression* const cexp1 = new ConstantExpression(exp1->get_position(), exp1);
902  eat();
903  ConstantExpression* const cexp2 = parse_constant_expression(table);
904  match(tok::RBRACK, REPAIR);
905  Slice* const slice = new Slice(start, ident.get_lexeme(), cexp1, cexp2, table);
906  return parse_procedural_assignment(slice, table);
907  } else {
908  match(tok::RBRACK, REPAIR);
909  Index* const index = new Index(start, ident.get_lexeme(), exp1, table);
910  return parse_procedural_assignment(index, table);
911  }
912  } else {
913  Identifier* const identifier = new Identifier(ident, table);
914  return parse_procedural_assignment(identifier, table);
915  }
916  } else {
917  Token matched = peek();
918  error_log.add(Error("Unexpected Statement token of type " + type_to_string(matched.get_type()) + " and lexeme " + matched.get_lexeme() + " found", matched.get_position()));
919  error_log.print();
920  exit(1);
921  return NULL;
922  }
923 }
